#include <stdio.h>
#include <stdlib.h>

int main(void) {

	float matrix_1[2][2],matrix_2[2][2];
	printf("Enter the elements of 1st matrix \n");
	int i,j;
	for(i=0;i<2;i++)
	{
		for(j=0;j<2;j++)
		{
			printf("Enter a%d%d: ",i+1,j+1);
			fflush(stdin); fflush(stdout);
			scanf("%f",&matrix_1[i][j]);
		}

	}
	printf("Enter the elements of 2st matrix \n");
	for(i=0;i<2;i++)
	{
		for(j=0;j<2;j++)
		{
			printf("Enter b%d%d: ",i+1,j+1);
			fflush(stdin); fflush(stdout);
			scanf("%f",&matrix_2[i][j]);
		}

	}
	printf("\nSum of Matrix: \n");
	for(i=0;i<2;i++)
		{
			for(j=0;j<2;j++)
			{
				printf("%0.2f \t",matrix_1[i][j]+matrix_2[i][j]);
			}
			printf("\n");

		}
}

#include <stdio.h>
#include <stdlib.h>

int main(void) {

	int in_array;
	printf("Enter the number of data: ");
	fflush(stdin);fflush(stdout);
	scanf("%d",&in_array);
	float array[in_array];
	int i;
	for(i=0; i<in_array ;i++ )
	{
		printf("%d. Enter number: ",i+1);
		fflush(stdin); fflush(stdout);
		scanf("%f",&array[i]);
	}
	float Average =0;
	for(i=0; i<in_array ;i++ )
		{
			Average += array[i];
		}
	printf("Average = %0.2f",Average/in_array);

}

#include <stdio.h>
#include <stdlib.h>

int main(void) {

	int row, column;

	printf("Enter rows and column of matrix: ");
	fflush(stdin);fflush(stdout);
	scanf("%d %d",&row,&column);

	float array[row][column],transpose[column][row];

	printf("Enter elements of matrix:\n");
	int i,j;
	for(i=0;i<row;i++)
	{
		for(j=0;j<column;j++)
		{
			printf("Enter elements a%d%d: ",i+1,j+1);
			fflush(stdin);fflush(stdout);
			scanf("%f",&array[i][j]);
		}
	}

	printf("Entered Matrix: \n");

	for(i=0;i<row;i++)
	{
		for(j=0;j<column;j++)
		{
			printf("%0.1f \t",array[i][j]);
		}
		printf("\n");
	}

	printf("Transpose of Matrix: \n");

	for(i=0;i<column;i++)
	{
		for(j=0;j<row;j++)
		{
			transpose[i][j]=array[j][i];
		}
	}
	for(i=0;i<column;i++)
	{
		for(j=0;j<row;j++)
		{
			printf("%0.1f \t",transpose[i][j]);
		}
		printf("\n");
	}
}
